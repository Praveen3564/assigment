//write a program to count no.of vowels  present in a string and also count no.of constants present in the string.

package string;

import java.util.*;

public class String2 
{
	public static void main(String[] args) 
	{
		int vowels=0;
		int  consonents=0;
		//Enter String....
		@SuppressWarnings("resource")
		Scanner st_add=new Scanner(System.in);
		System.out.println("Enter the String :");
		String s=st_add.next();
		for (int a=0;a<s.length();a++)
		{
			char c=s.charAt(a);
			if ((c=='A')||(c=='E')||(c=='I')||(c=='O')||(c=='U')||(c=='a')||(c=='e')||(c=='i')||(c=='o')||(c=='u'))
			{
				vowels++;
			}
			else
			{
				consonents++;
			}
		}
		System.out.println("There are "+vowels+" number of vowels in the given string.");
		System.out.println("There are "+consonents+" number of consonents in the given string.");
	}

}

//Write a program to check whether given String is palindrome or not.

package string;

import java.util.Scanner;

public class String7 
{
	public static void main(String[] args) 
	{
		@SuppressWarnings("resource")
		Scanner scan=new Scanner(System.in);
		System.out.println("Enter the String : ");
		String s1=scan.nextLine();
		String s2="";
		for (int i=s1.length()-1;i>=0;i--)
		{
			s2=s2+s1.charAt(i);
		}
		if(s1.equalsIgnoreCase(s2
				))
		{
			System.out.println("The given String is Palindrom");
		}
		else
		{
			System.out.println("Given String is not a Palindrom");
		}
	}
}

package string;

import java.util.Scanner;

public class String11 
{
	public static void main(String[] args) 
	{
		@SuppressWarnings("resource")
		Scanner scan=new Scanner(System.in);
		System.out.println("Enter the String: ");
		String s=scan.nextLine();
		String[] str=s.split(" ");
		String s1="";
		String s2="";;
		for (int i=0;i<=str.length-1;i++)
		{
			s1=str[i];
			for (int j=s1.length()-1;j>=0;j--)
			{
				char  ch=s1.charAt(j);
				s2=s2+ch;
			}
			s2=s2+" ";
		}
		System.out.println(s2);
	}
}
